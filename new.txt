        int amount = 0;
        int n = 36;
        while (n < 45) {

            ItemStack itemStack;
            if ((itemStack = MinecraftInstance.mc.thePlayer.inventoryContainer.getSlot(i = n++).getStack()) == null || !(itemStack.getItem() instanceof ItemBlock)) continue;
            Item item = itemStack.getItem();
            if (item == null) {
                throw new NullPointerException("null cannot be cast to non-null type net.minecraft.item.ItemBlock");
            }
            Block block = ((ItemBlock)item).block;
            Intrinsics.checkNotNullExpressionValue(block, "itemStack.item as ItemBlock).block");

            amount += itemStack.stackSize;
        }

    }
